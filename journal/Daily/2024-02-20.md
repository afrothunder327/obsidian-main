  .fill(start)
  .map((year, index) => year + index);

const availableYears = rangeOfYears(minYear, currentYear - 1);
const month = ("0" + (d.month() + 1)).slice(-2);

	var note = dv.page(`${journalPath}/${y}-${dateString}`);
	if (note) {

});
